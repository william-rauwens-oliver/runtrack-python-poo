import random

class Personnage:
    def __init__(self, nom, vie):
        self.nom = nom
        self.vie = vie

    def attaquer(self, adversaire, points):
        print(f"{self.nom} attaque {adversaire.nom} et lui enlève {points} points de vie.")
        adversaire.vie -= points

class Jeu:
    def __init__(self):
        self.niveau = 1

    def choisirNiveau(self):
        self.niveau = int(input("Choisissez le niveau de difficulté (1, 2, ou 3) : "))

    def lancerJeu(self):
        self.choisirNiveau()

        points_joueur = self.niveau * 10
        points_ennemi = self.niveau * 10

        joueur = Personnage("Joueur", points_joueur)
        ennemi = Personnage("Ennemi", points_ennemi)

        print(f"Le combat commence ! Niveau de difficulté : {self.niveau}")
        self.deroulementPartie(joueur, ennemi)

    def verifierSante(self, personnage):
        if personnage.vie <= 0:
            print(f"{personnage.nom} n'a plus de vie.")

    def verifierGagnant(self, joueur, ennemi):
        if joueur.vie <= 0:
            print("Vous avez perdu. L'ennemi a gagné.")
        elif ennemi.vie <= 0:
            print("Félicitations ! Vous avez vaincu l'ennemi.")
        else:
            return False
        return True

    def deroulementPartie(self, joueur, ennemi):
        while True:
            joueur.attaque(ennemi, 5)
            self.verifierSante(ennemi)

            ennemi_attaque_points = random.randint(1, 5)  # Attaque aléatoire entre 1 et 5 points
            ennemi.attaque(joueur, ennemi_attaque_points)
            self.verifierSante(joueur)

            if self.verifierGagnant(joueur, ennemi):
                break

# Exécution du jeu
jeu = Jeu()
jeu.lancerJeu()
