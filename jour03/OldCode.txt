class Personnage:
    def __init__(self, nom, vie):
        self.nom = nom
        self.vie = vie

    def attaquer(self, adversaire, degats):
        print(f"{self.nom} attaque {adversaire.nom} et lui inflige {degats} points de dégâts.")
        adversaire.vie -= degats

class Jeu:
    def __init__(self):
        self.niveau = 1

    def choisirNiveau(self):
        self.niveau = int(input("Choisissez le niveau de difficulté (1 facile, 2 moyen, 3 difficile) : "))

    def lancerJeu(self):
        self.choisirNiveau()

        niveaux = {
            1: {"vie_joueur": 50, "vie_ennemi": 30, "degats_joueur": 5, "degats_ennemi": 3},
            2: {"vie_joueur": 40, "vie_ennemi": 40, "degats_joueur": 4, "degats_ennemi": 4},
            3: {"vie_joueur": 30, "vie_ennemi": 50, "degats_joueur": 3, "degats_ennemi": 5}
        }

        niveau = niveaux.get(self.niveau, niveaux[1])

        joueur = Personnage("Joueur", niveau["vie_joueur"])
        ennemi = Personnage("Ennemi", niveau["vie_ennemi"])

        while joueur.vie > 0 and ennemi.vie > 0:
            joueur.attaquer(ennemi, niveau["degats_joueur"])
            ennemi.attaquer(joueur, niveau["degats_ennemi"])

            print(f"{joueur.nom} - Vie: {joueur.vie} | {ennemi.nom} - Vie: {ennemi.vie}\n")

        if joueur.vie <= 0:
            print("Vous avez perdu. L'ennemi a gagné.")
        else:
            print("Félicitations ! Vous avez vaincu l'ennemi.")

# Exemple d'utilisation
jeu = Jeu()
jeu.lancerJeu()
