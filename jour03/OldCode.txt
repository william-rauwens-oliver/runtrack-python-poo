# Création d'une première instance de CompteBancaire
compte1 = CompteBancaire(numero_compte="12345", nom="Dupont", prenom="Jean", solde=1000)

# Affichage des détails du compte
compte1.afficher()

# Test des différentes méthodes
compte1.versement(500)
compte1.retrait(200)
compte1.agios(taux_agios=0.02)

# Création d'une deuxième instance de CompteBancaire à découvert
compte2 = CompteBancaire(numero_compte="67890", nom="Martin", prenom="Sophie", solde=-200, decouvert=True)

# Affichage des détails du deuxième compte
compte2.afficher()

# Virement du premier compte vers le deuxième pour le remettre à zéro
compte1.virement(compte_destinataire=compte2, montant=1200)
